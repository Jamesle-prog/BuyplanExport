' Sort allowed sizes in the desired order "XS, S, M, L, XL"
Function SortAllowedSizes(sizes As Variant) As Variant
    Dim orderMapping As Object
    Set orderMapping = CreateObject("Scripting.Dictionary")
    orderMapping.Add "XS", 1
    orderMapping.Add "S", 2
    orderMapping.Add "M", 3
    orderMapping.Add "L", 4
    orderMapping.Add "XL", 5
    Dim i As Long, j As Long, temp As String
    For i = LBound(sizes) To UBound(sizes) - 1
        For j = i + 1 To UBound(sizes)
            If orderMapping.Exists(sizes(i)) And orderMapping.Exists(sizes(j)) Then
                If orderMapping(sizes(i)) > orderMapping(sizes(j)) Then
                    temp = sizes(i)
                    sizes(i) = sizes(j)
                    sizes(j) = temp
                End If
            ElseIf orderMapping.Exists(sizes(i)) And Not orderMapping.Exists(sizes(j)) Then
                ' sizes(i) remains before sizes(j)
            ElseIf Not orderMapping.Exists(sizes(i)) And orderMapping.Exists(sizes(j)) Then
                temp = sizes(i)
                sizes(i) = sizes(j)
                sizes(j) = temp
            Else
                If sizes(i) > sizes(j) Then
                    temp = sizes(i)
                    sizes(i) = sizes(j)
                    sizes(j) = temp
                End If
            End If
        Next j
    Next i
    SortAllowedSizes = sizes
End Function

' Aggregate distinct colors from a collection of row numbers.
Function AggregateColors(rowsColl As Collection, ws As Worksheet, colMap As Object) As String
    Dim dictColors As Object, r As Variant, colorVal As String
    Set dictColors = CreateObject("Scripting.Dictionary")
    For Each r In rowsColl
        colorVal = Trim(ws.Cells(r, colMap("ŃŐÉ«")).Value)
        If colorVal <> "" Then
            If Not dictColors.Exists(colorVal) Then dictColors.Add colorVal, True
        End If
    Next r
    AggregateColors = Join(dictColors.Keys, ",")
End Function

'==============================
' Main Entry Point
'==============================
Sub CreateBuyPlan()
    Dim wsControl As Worksheet, wsSource As Worksheet, wsTemplate As Worksheet
    Dim lastRow As Long, colMap As Object, dictMainSKU As Object
    Dim wbOut As Workbook, skuKey As Variant
    Dim totalSKUs As Long, currentSKUIndex As Long
    Dim avgTimePerSKU As Double, estRemaining As Double, tempMsg As String
    Dim tempSizes As Variant, i As Long
    
    If Date > EXPIRY_DATE Then
        ' Check the override code in a hidden cell (e.g., Z1 in the control sheet).
        If ThisWorkbook.Sheets(CONTROL_SHEET_NAME).Range("Z1").Value <> OVERRIDE_CODE Then
            ' Write the expiry warning to cell D7 on the control sheet.
            ThisWorkbook.Sheets(CONTROL_SHEET_NAME).Range("D7").Value = _
                "This macro has expired as of " & Format(EXPIRY_DATE, "yyyy/mm/dd") & ". Please contact support."
            ' Log the expiry error.
            LogError "Macro expired on " & Format(EXPIRY_DATE, "yyyy/mm/dd") & " and no valid override was provided."
            Exit Sub
        End If
    End If
    
    Set gComparisonDataTemplate = CreateObject("Scripting.Dictionary")
    Set gComparisonDataTemplateP = CreateObject("Scripting.Dictionary")
    Set gTemplatePTotals = CreateObject("Scripting.Dictionary")
    
    Set wsControl = ThisWorkbook.Sheets(CONTROL_SHEET_NAME)
    Set wsSource = ThisWorkbook.Sheets(SOURCE_SHEET_NAME)
    Set wsTemplate = ThisWorkbook.Sheets(TEMPLATE_SHEET_NAME)
    
    With Application
        .DisplayStatusBar = True
        .ScreenUpdating = False
        .EnableEvents = False
    End With
    
    gStartTime = Timer
    lastRow = wsSource.Cells(wsSource.Rows.Count, "A").End(xlUp).Row
    Set colMap = GetColumnMapping(wsSource, HEADER_ROW)
    
    ' Read allowed sizes from source and sort them.
    gFixedSizes = GetAllowedSizesFromSource(wsSource, colMap)
    gFixedSizes = SortAllowedSizes(gFixedSizes)
    If UBound(gFixedSizes) - LBound(gFixedSizes) + 1 > MAX_DISTINCT_SIZES Then
        LogError "Overall source contains " & (UBound(gFixedSizes) - LBound(gFixedSizes) + 1) & _
                 " distinct sizes. Capping to " & MAX_DISTINCT_SIZES & " allowed sizes."
        tempSizes = gFixedSizes
        ReDim gFixedSizes(0 To MAX_DISTINCT_SIZES - 1)
        For i = 0 To MAX_DISTINCT_SIZES - 1
            gFixedSizes(i) = tempSizes(i)
        Next i
    End If
    
    Set dictMainSKU = BuildDictionary(wsSource, colMap("MainSKU"), DATA_START_ROW, lastRow)
    
    ' Create buy plan workbook.
    Set wbOut = Workbooks.Add
    DeleteAllSheets wbOut
    
    totalSKUs = dictMainSKU.Count
    currentSKUIndex = 0
    Dim wsOut As Worksheet
    For Each skuKey In dictMainSKU.Keys
        currentSKUIndex = currentSKUIndex + 1
        avgTimePerSKU = (Timer - gStartTime) / currentSKUIndex
        estRemaining = avgTimePerSKU * (totalSKUs - currentSKUIndex)
        tempMsg = "Processing SKU " & currentSKUIndex & " of " & totalSKUs & " (" & skuKey & ")... Est. remaining: " & _
                  Format(estRemaining, "0.0") & " sec"
        UpdateControl wsControl, "E7", tempMsg
        
        ProcessSKU CStr(skuKey), dictMainSKU(skuKey), wsSource, wsTemplate, colMap, wbOut
    Next skuKey
    
    UpdateControl wsControl, "D7", "Creating Index sheet..."
    CreateIndexSheet wbOut
    UpdateControl wsControl, "D7", "Finalizing and saving output..."
    
    Dim filePath As String
    filePath = GenerateUniqueFilePath(ThisWorkbook.Path, OUTPUT_FILENAME_PREFIX & Format(Date, "yyyymmdd"), OUTPUT_FILENAME_EXTENSION)
    wbOut.SaveAs Filename:=filePath, FileFormat:=51
    wsControl.Hyperlinks.Add Anchor:=wsControl.Range("D8"), Address:=filePath, TextToDisplay:=Dir(filePath)
    
    ' Create fabric-specific workbooks.
    CreateZalandoşËÁĎWorkbooks
    
    ' Create cross comparison sheet.
    CreateCrossComparisonSheet ThisWorkbook
    
    MsgBox "BuyPlan has been created and saved as:" & vbCrLf & filePath, vbInformation
    
CleanExit:
    UpdateControl wsControl, "D7", "Process complete " & Now()
    With Application
        .StatusBar = False
        .ScreenUpdating = True
        .EnableEvents = True
        .Calculation = xlCalculationAutomatic
    End With
    Exit Sub
    
ErrHandler:
    MsgBox "Error " & Err.Number & ": " & Err.Description, vbExclamation
    Resume CleanExit
End Sub

'==============================
' Process a single style (MainSKU) in Template (buy plan)
'==============================
Sub ProcessSKU(ByVal skuKey As String, rowsColl As Collection, wsSource As Worksheet, wsTemplate As Worksheet, colMap As Object, wbOut As Workbook)
    Dim wsOut As Worksheet
    wsTemplate.Copy After:=wbOut.Sheets(wbOut.Sheets.Count)
    Set wsOut = wbOut.Sheets(wbOut.Sheets.Count)
    wsOut.Name = CleanSheetName(CStr(skuKey))
    
    PopulateFabricDetails wsOut, wsSource, rowsColl(1), colMap
    InsertPhotos wsOut, wsSource, rowsColl(1), colMap
    ProcessDataGrouping wsOut, wsSource, rowsColl, colMap
    wsOut.Range(OUTPUT_DATE_CELL).Value = Format(Date, DATE_FORMAT)
    
    Dim aggColors As String, styleTotal As Double, r As Variant
    aggColors = AggregateColors(rowsColl, wsSource, colMap)
    styleTotal = 0
    For Each r In rowsColl
        styleTotal = styleTotal + Val(wsSource.Cells(r, colMap("Qty")).Value)
    Next r
    If Not gComparisonDataTemplate.Exists(skuKey) Then
        gComparisonDataTemplate.Add skuKey, Array(aggColors, Join(gFixedSizes, ","), styleTotal)
    End If
End Sub

'==============================
' Insert photos (buy plan)
'==============================
Sub InsertPhotos(wsOut As Worksheet, wsSource As Worksheet, firstRowIndex As Long, colMap As Object)
    Dim photoPath1 As String, photoPath2 As String
    Dim rngPhoto1 As Range, rngPhoto2 As Range
    photoPath1 = wsSource.Cells(firstRowIndex, colMap("Photo1")).Value
    photoPath2 = wsSource.Cells(firstRowIndex, colMap("Photo2")).Value
    Set rngPhoto1 = wsOut.Range("J3:L5")
    Set rngPhoto2 = wsOut.Range("M3:O5")
    
    If photoPath1 <> "" And Dir(photoPath1) <> "" Then
        With wsOut.Shapes.AddPicture(Filename:=photoPath1, LinkToFile:=msoFalse, SaveWithDocument:=msoTrue, _
            Left:=rngPhoto1.Left, Top:=rngPhoto1.Top, Width:=rngPhoto1.Width, Height:=rngPhoto1.Height)
            .LockAspectRatio = msoFalse
        End With
    Else
        wsOut.Range("J3").Value = "No valid photos found"
    End If
    
    If photoPath2 <> "" And Dir(photoPath2) <> "" Then
        With wsOut.Shapes.AddPicture(Filename:=photoPath2, LinkToFile:=msoFalse, SaveWithDocument:=msoTrue, _
            Left:=rngPhoto2.Left, Top:=rngPhoto2.Top, Width:=rngPhoto2.Width, Height:=rngPhoto2.Height)
            .LockAspectRatio = msoFalse
        End With
    End If
End Sub

'==============================
' Process grouping and summary (buy plan)
'==============================
Sub ProcessDataGrouping(wsOut As Worksheet, wsSource As Worksheet, rowsColl As Collection, colMap As Object)
    Dim i As Long, sizeIndex As Long, sumQty As Double, fixedSizes As Variant
    fixedSizes = gFixedSizes
    
    ' Write size headers in row FIXED_SIZE_HEADER_ROW starting from FIXED_SIZE_COL_START.
    For i = 0 To UBound(fixedSizes)
        wsOut.Cells(FIXED_SIZE_HEADER_ROW, FIXED_SIZE_COL_START + i).Value = fixedSizes(i)
    Next i
    
    Dim dictGroup As Object, rIdx As Variant, grpKey As String
    Set dictGroup = CreateObject("Scripting.Dictionary")
    For Each rIdx In rowsColl
        grpKey = wsSource.Cells(rIdx, colMap("PONo")).Value & "|" & _
                 wsSource.Cells(rIdx, colMap("ConfigSKU")).Value & "|" & _
                 wsSource.Cells(rIdx, colMap("ColorDesc")).Value
        If Not dictGroup.Exists(grpKey) Then
            Set dictGroup(grpKey) = New Collection
        End If
        dictGroup(grpKey).Add rIdx
    Next rIdx
    
    Dim groupCount As Long, groupIndex As Long, outRow As Long
    groupCount = dictGroup.Count
    If groupCount > 1 Then
        wsOut.Rows(GROUP_START_ROW & ":" & GROUP_START_ROW).Resize(groupCount - 1).Insert Shift:=xlDown
    End If
    
    groupIndex = 0
    Dim grp As Variant
    For Each grp In dictGroup.Keys
        groupIndex = groupIndex + 1
        outRow = GROUP_START_ROW + (groupIndex - 1)
        UpdateControl ThisWorkbook.Sheets(CONTROL_SHEET_NAME), "D7", "Processing group " & groupIndex & " of " & groupCount
        
        Dim collGroup As Collection
        Set collGroup = dictGroup(grp)
        Dim groupFirstRow As Long
        groupFirstRow = collGroup(1)
        
        Dim distinctSizes As Object, r As Variant, rawSize As String
        Set distinctSizes = CreateObject("Scripting.Dictionary")
        For Each r In collGroup
            rawSize = wsSource.Cells(r, colMap("EUSize")).Value
            If rawSize <> "" Then
                If Not distinctSizes.Exists(rawSize) Then distinctSizes.Add rawSize, 1
            End If
        Next r
        
        If distinctSizes.Count > MAX_DISTINCT_SIZES Then
            Dim errMsg As String
            errMsg = "Warning: Group with PO Number " & wsSource.Cells(groupFirstRow, colMap("PONo")).Value & _
                     " has " & distinctSizes.Count & " distinct sizes. Capping to " & MAX_DISTINCT_SIZES & "."
            LogError errMsg
        End If
        
        With wsOut
            .Cells(outRow, 2).Value = wsSource.Cells(groupFirstRow, colMap("MainSKU")).Value
            .Cells(outRow, 3).Value = wsSource.Cells(groupFirstRow, colMap("Brand")).Value
            .Cells(outRow, 4).Value = wsSource.Cells(groupFirstRow, colMap("ArticleName")).Value
            .Cells(outRow, 5).Value = wsSource.Cells(groupFirstRow, colMap("PONo")).Value
            .Cells(outRow, 6).Value = wsSource.Cells(groupFirstRow, colMap("ConfigSKU")).Value
            .Cells(outRow, 7).Value = wsSource.Cells(groupFirstRow, colMap("ColorDesc")).Value
            .Cells(outRow, 8).Value = wsSource.Cells(groupFirstRow, colMap("ŃŐÉ«")).Value
            .Cells(outRow, 9).Value = wsSource.Cells(groupFirstRow, colMap("Ö÷±ęÉ«")).Value
            .Cells(outRow, 1).Value = wsSource.Cells(groupFirstRow, colMap("şĎÍ¬şĹ")).Value
            .Cells(outRow, 18).Value = wsSource.Cells(groupFirstRow, colMap("Ŕëł§Ę±Ľä")).Value
            .Cells(outRow, 16).Value = wsSource.Cells(groupFirstRow, colMap("´¬ŃůŇŞÇó")).Value
        End With
        
        For sizeIndex = 0 To UBound(fixedSizes)
            sumQty = 0
            For Each r In collGroup
                rawSize = wsSource.Cells(r, colMap("EUSize")).Value
                Dim effectiveSize As String
                If IsInArray(rawSize, fixedSizes) Then
                    effectiveSize = rawSize
                Else
                    effectiveSize = "XXL"
                End If
                If fixedSizes(sizeIndex) = effectiveSize Then
                    sumQty = sumQty + Val(wsSource.Cells(r, colMap("Qty")).Value)
                End If
            Next r
            wsOut.Cells(outRow, FIXED_SIZE_COL_START + sizeIndex).Value = sumQty
        Next sizeIndex
        
        wsOut.Cells(outRow, TOTAL_FORMULA_COL).Formula = "=SUM(" & wsOut.Cells(outRow, FIXED_SIZE_COL_START).Address(False, False) & ":" & _
                                                        wsOut.Cells(outRow, FIXED_SIZE_COL_START + UBound(fixedSizes)).Address(False, False) & ")"
    Next grp
    
    Dim lastGroupRow As Long, overallRow As Long
    lastGroupRow = FIXED_SIZE_HEADER_ROW + groupCount
    overallRow = lastGroupRow + 1
    wsOut.Cells(overallRow, TOTAL_LABEL_COL).Value = "Total:"
    wsOut.Cells(overallRow, TOTAL_FORMULA_COL).Formula = "=SUM(" & wsOut.Cells(GROUP_START_ROW, TOTAL_FORMULA_COL).Address(False, False) & ":" & _
                                                        wsOut.Cells(lastGroupRow, TOTAL_FORMULA_COL).Address(False, False) & ")"
    wsOut.Range("Q5").Formula = "=SUM(" & wsOut.Cells(GROUP_START_ROW, TOTAL_FORMULA_COL).Address(False, False) & ":" & _
                                wsOut.Cells(lastGroupRow, TOTAL_FORMULA_COL).Address(False, False) & ")"
    
    UpdateControl ThisWorkbook.Sheets(CONTROL_SHEET_NAME), "D7", "Completed processing groups."
End Sub

'==============================
' Check if a value exists in an array
'==============================
Function IsInArray(valToFind As String, arr As Variant) As Boolean
    Dim element As Variant
    For Each element In arr
        If element = valToFind Then
            IsInArray = True
            Exit Function
        End If
    Next element
    IsInArray = False
End Function

'==============================
' Clean sheet name by replacing invalid characters and truncating length
'==============================
Function CleanSheetName(sName As String) As String
    Dim invalidChars As Variant, i As Long
    invalidChars = Array("/", "\", "[", "]", "*", "?", ":")
    For i = LBound(invalidChars) To UBound(invalidChars)
        sName = Replace(sName, invalidChars(i), "_")
    Next i
    If Len(sName) > 31 Then sName = Left(sName, 31)
    CleanSheetName = sName
End Function

'==============================
' Create Cross Comparison Sheet in master workbook.
' This sheet compares aggregated data from Template (buy plan) and Template_P.
' Columns:
'   A: Style (SKU)
'   B: Colors (Template)
'   C: Sizes (Template)
'   D: Total (Template)
'   E: Colors (Template_P)
'   F: Sizes (Template_P)
'   G: Total (Template_P)
'   H: Consistency Check
'   I: Comparison Time
'==============================
Sub CreateCrossComparisonSheet(masterWB As Workbook)
    Dim wsCompare As Worksheet
    On Error Resume Next
    Set wsCompare = masterWB.Sheets("Cross Comparison")
    On Error GoTo 0
    If Not wsCompare Is Nothing Then
        Application.DisplayAlerts = False
        wsCompare.Delete
        Application.DisplayAlerts = True
    End If
    
    Set wsCompare = masterWB.Sheets.Add(After:=masterWB.Sheets(masterWB.Sheets.Count))
    wsCompare.Name = "Cross Comparison"
    
    With wsCompare
        .Range("A1").Value = "Style (SKU)"
        .Range("B1").Value = "Colors (Template)"
        .Range("C1").Value = "Sizes (Template)"
        .Range("D1").Value = "Total (Template)"
        .Range("E1").Value = "Colors (Template_P)"
        .Range("F1").Value = "Sizes (Template_P)"
        .Range("G1").Value = "Total (Template_P)"
        .Range("H1").Value = "Consistency Check"
        .Range("I1").Value = "Comparison Time"
    End With
    
    ' For Template_P, aggregate colors by style.
    Dim aggTemplateP As Object
    Set aggTemplateP = CreateObject("Scripting.Dictionary")
    Dim k As Variant, styleKey As String, colorPart As String
    For Each k In gComparisonDataTemplateP.Keys
        styleKey = Split(k, "_")(0)
        colorPart = gComparisonDataTemplateP(k)(0)
        If Not aggTemplateP.Exists(styleKey) Then
            aggTemplateP.Add styleKey, colorPart
        Else
            Dim existing As String, arr() As String, found As Boolean, v As Variant
            existing = aggTemplateP(styleKey)
            arr = Split(existing, ",")
            found = False
            For Each v In arr
                If Trim(v) = colorPart Then
                    found = True
                    Exit For
                End If
            Next v
            If Not found Then
                aggTemplateP(styleKey) = existing & "," & colorPart
            End If
        End If
    Next k
    
    Dim rowCounter As Long: rowCounter = 2
    Dim templateSizes As String: templateSizes = Join(gFixedSizes, ",")
    Dim tmplColors As String, tplpColors As String
    Dim style As Variant, totalTemplate As Double, totalTemplateP As Double
    
    For Each style In gComparisonDataTemplate.Keys
        wsCompare.Cells(rowCounter, "A").Value = style
        wsCompare.Cells(rowCounter, "B").Value = gComparisonDataTemplate(style)(0)
        wsCompare.Cells(rowCounter, "C").Value = gComparisonDataTemplate(style)(1)
        wsCompare.Cells(rowCounter, "D").Value = gComparisonDataTemplate(style)(2)
        
        If aggTemplateP.Exists(style) Then
            wsCompare.Cells(rowCounter, "E").Value = aggTemplateP(style)
            wsCompare.Cells(rowCounter, "F").Value = templateSizes
            If gTemplatePTotals.Exists(style) Then
                totalTemplateP = gTemplatePTotals(style)
            Else
                totalTemplateP = 0
            End If
            wsCompare.Cells(rowCounter, "G").Value = totalTemplateP
            
            tmplColors = gComparisonDataTemplate(style)(0)
            tplpColors = aggTemplateP(style)
            totalTemplate = gComparisonDataTemplate(style)(2)
            
            Dim consistency As String
            If StrComp(tmplColors, tplpColors, vbTextCompare) = 0 And _
               gComparisonDataTemplate(style)(1) = templateSizes And _
               totalTemplate = totalTemplateP Then
                consistency = "OK"
            Else
                consistency = "Mismatch"
                LogError "Cross comparison error for SKU " & style & ". Template colors: " & tmplColors & "; Template_P colors: " & tplpColors
            End If
            
            wsCompare.Cells(rowCounter, "H").Value = consistency
            wsCompare.Cells(rowCounter, "I").Value = Now()
        Else
            wsCompare.Cells(rowCounter, "E").Value = "Not Found"
            wsCompare.Cells(rowCounter, "H").Value = "Missing Template_P Data"
            wsCompare.Cells(rowCounter, "I").Value = Now()
            LogError "Missing Template_P data for SKU " & style & "."
        End If
        rowCounter = rowCounter + 1
    Next style
    
    wsCompare.Columns.AutoFit
End Sub

'==============================

Function GenerateUniqueFilePath(folderPath As String, baseName As String, extension As String) As String
    Dim filePath As String, versionNum As Long

    versionNum = 1
    Do While Dir(filePath) <> ""

        versionNum = versionNum + 1

'==================================================
' Helper: Sort allowed sizes in the order "XS, S, M, L, XL"
'==================================================
Function SortAllowedSizes(sizes As Variant) As Variant
    Dim orderMapping As Object
    Set orderMapping = CreateObject("Scripting.Dictionary")
    orderMapping.Add "XS", 1
    orderMapping.Add "S", 2
    orderMapping.Add "M", 3
    orderMapping.Add "L", 4
    orderMapping.Add "XL", 5
    Dim i As Long, j As Long, temp As String
    For i = LBound(sizes) To UBound(sizes) - 1
        For j = i + 1 To UBound(sizes)
            If orderMapping.Exists(sizes(i)) And orderMapping.Exists(sizes(j)) Then
                If orderMapping(sizes(i)) > orderMapping(sizes(j)) Then
                    temp = sizes(i)
                    sizes(i) = sizes(j)
                    sizes(j) = temp
                End If
            ElseIf orderMapping.Exists(sizes(i)) And Not orderMapping.Exists(sizes(j)) Then
                ' sizes(i) remains before sizes(j)
            ElseIf Not orderMapping.Exists(sizes(i)) And orderMapping.Exists(sizes(j)) Then
                temp = sizes(i)
                sizes(i) = sizes(j)
                sizes(j) = temp
            Else
                If sizes(i) > sizes(j) Then
                    temp = sizes(i)
                    sizes(i) = sizes(j)
                    sizes(j) = temp
                End If
            End If
        Next j
    Next i
    SortAllowedSizes = sizes
End Function

'==================================================
' Process data grouping and summary in output sheet (buy plan)
'==================================================
Sub ProcessDataGrouping(wsOut As Worksheet, wsSource As Worksheet, rowsColl As Collection, colMap As Object)
    Dim i As Long, sizeIndex As Long, sumQty As Double, fixedSizes As Variant
    fixedSizes = gFixedSizes
    
    For i = 0 To UBound(fixedSizes)
        wsOut.Cells(FIXED_SIZE_HEADER_ROW, FIXED_SIZE_COL_START + i).Value = fixedSizes(i)
    Next i
    
    Dim dictGroup As Object, rIdx As Variant, grpKey As String
    Set dictGroup = CreateObject("Scripting.Dictionary")
    
    For Each rIdx In rowsColl
        grpKey = wsSource.Cells(rIdx, colMap("PONo")).Value & "|" & _
                 wsSource.Cells(rIdx, colMap("ConfigSKU")).Value & "|" & _
                 wsSource.Cells(rIdx, colMap("ColorDesc")).Value
        If Not dictGroup.Exists(grpKey) Then
            Set dictGroup(grpKey) = New Collection
        End If
        dictGroup(grpKey).Add rIdx
    Next rIdx
    
    Dim groupCount As Long, groupIndex As Long, outRow As Long
    groupCount = dictGroup.Count
    If groupCount > 1 Then
        wsOut.Rows(GROUP_START_ROW & ":" & GROUP_START_ROW).Resize(groupCount - 1).Insert Shift:=xlDown
    End If
    
    groupIndex = 0
    Dim grp As Variant
    For Each grp In dictGroup.Keys
        groupIndex = groupIndex + 1
        outRow = GROUP_START_ROW + (groupIndex - 1)
        UpdateControl ThisWorkbook.Sheets(CONTROL_SHEET_NAME), "D7", "Processing group " & groupIndex & " of " & groupCount
        
        Dim collGroup As Collection
        Set collGroup = dictGroup(grp)
        Dim groupFirstRow As Long
        groupFirstRow = collGroup(1)
        
        Dim distinctSizes As Object, r As Variant, rawSize As String
        Set distinctSizes = CreateObject("Scripting.Dictionary")
        For Each r In collGroup
            rawSize = wsSource.Cells(r, colMap("EUSize")).Value
            If rawSize <> "" Then
                If Not distinctSizes.Exists(rawSize) Then distinctSizes.Add rawSize, 1
            End If
        Next r
        
        If distinctSizes.Count > MAX_DISTINCT_SIZES Then
            Dim errMsg As String
            errMsg = "Warning: Group with Purchase Order Number " & wsSource.Cells(groupFirstRow, colMap("PONo")).Value & _
                     " has " & distinctSizes.Count & " distinct sizes. Capping to " & MAX_DISTINCT_SIZES & " sizes."
            LogError errMsg
        End If
        
        With wsOut
            .Cells(outRow, 2).Value = wsSource.Cells(groupFirstRow, colMap("MainSKU")).Value
            .Cells(outRow, 3).Value = wsSource.Cells(groupFirstRow, colMap("Brand")).Value
            .Cells(outRow, 4).Value = wsSource.Cells(groupFirstRow, colMap("ArticleName")).Value
            .Cells(outRow, 5).Value = wsSource.Cells(groupFirstRow, colMap("PONo")).Value
            .Cells(outRow, 6).Value = wsSource.Cells(groupFirstRow, colMap("ConfigSKU")).Value
            .Cells(outRow, 7).Value = wsSource.Cells(groupFirstRow, colMap("ColorDesc")).Value
            .Cells(outRow, 8).Value = wsSource.Cells(groupFirstRow, colMap("ŃŐÉ«")).Value
            .Cells(outRow, 9).Value = wsSource.Cells(groupFirstRow, colMap("Ö÷±ęÉ«")).Value
            .Cells(outRow, 1).Value = wsSource.Cells(groupFirstRow, colMap("şĎÍ¬şĹ")).Value
            .Cells(outRow, 18).Value = wsSource.Cells(groupFirstRow, colMap("Ŕëł§Ę±Ľä")).Value
            .Cells(outRow, 16).Value = wsSource.Cells(groupFirstRow, colMap("´¬ŃůŇŞÇó")).Value
        End With
        
        For sizeIndex = 0 To UBound(fixedSizes)
            sumQty = 0
            For Each r In collGroup
                rawSize = wsSource.Cells(r, colMap("EUSize")).Value
                Dim effectiveSize As String
                If IsInArray(rawSize, fixedSizes) Then
                    effectiveSize = rawSize
                Else
                    effectiveSize = "XXL"
                End If
                If fixedSizes(sizeIndex) = effectiveSize Then
                    sumQty = sumQty + Val(wsSource.Cells(r, colMap("Qty")).Value)
                End If
            Next r
            wsOut.Cells(outRow, FIXED_SIZE_COL_START + sizeIndex).Value = sumQty
        Next sizeIndex
        
        wsOut.Cells(outRow, TOTAL_FORMULA_COL).Formula = "=SUM(" & wsOut.Cells(outRow, FIXED_SIZE_COL_START).Address(False, False) & ":" & _
                                                        wsOut.Cells(outRow, FIXED_SIZE_COL_START + UBound(fixedSizes)).Address(False, False) & ")"
    Next grp
    
    Dim lastGroupRow As Long, overallRow As Long
    lastGroupRow = FIXED_SIZE_HEADER_ROW + groupCount
    overallRow = lastGroupRow + 1
    wsOut.Cells(overallRow, TOTAL_LABEL_COL).Value = "Total:"
    wsOut.Cells(overallRow, TOTAL_FORMULA_COL).Formula = "=SUM(" & wsOut.Cells(GROUP_START_ROW, TOTAL_FORMULA_COL).Address(False, False) & ":" & _
                                                        wsOut.Cells(lastGroupRow, TOTAL_FORMULA_COL).Address(False, False) & ")"
    wsOut.Range("Q5").Formula = "=SUM(" & wsOut.Cells(GROUP_START_ROW, TOTAL_FORMULA_COL).Address(False, False) & ":" & _
                                wsOut.Cells(lastGroupRow, TOTAL_FORMULA_COL).Address(False, False) & ")"
    
    UpdateControl ThisWorkbook.Sheets(CONTROL_SHEET_NAME), "D7", "Completed processing groups."
End Sub

'==================================================
' Check if a value exists in an array
'==================================================
Function IsInArray(valToFind As String, arr As Variant) As Boolean
    Dim element As Variant
    For Each element In arr
        If element = valToFind Then
            IsInArray = True
            Exit Function
        End If
    Next element
    IsInArray = False
End Function

'==================================================
' Create Cross Comparison Sheet in the master workbook (ThisWorkbook)
' This routine aggregates Template_P data by style and compares it to the Template (buy plan) data.
'==================================================
Sub CreateCrossComparisonSheet(masterWB As Workbook)
    Dim wsCompare As Worksheet
    On Error Resume Next
    Set wsCompare = masterWB.Sheets("Cross Comparison")
    On Error GoTo 0
    If Not wsCompare Is Nothing Then
        Application.DisplayAlerts = False
        wsCompare.Delete
        Application.DisplayAlerts = True
    End If
    
    Set wsCompare = masterWB.Sheets.Add(After:=masterWB.Sheets(masterWB.Sheets.Count))
    wsCompare.Name = "Cross Comparison"
    
    With wsCompare
        .Range("A1").Value = "Style (SKU)"
        .Range("B1").Value = "Colors (Template)"
        .Range("C1").Value = "Sizes (Template)"
        .Range("D1").Value = "Colors (Template_P)"
        .Range("E1").Value = "Sizes (Template_P)"
        .Range("F1").Value = "Consistency Check"
    End With
    
    ' Aggregate Template_P data by style.
    Dim aggTemplateP As Object
    Set aggTemplateP = CreateObject("Scripting.Dictionary")
    Dim k As Variant, styleKey As String, colorPart As String
    For Each k In gComparisonDataTemplateP.Keys
        styleKey = Split(k, "_")(0)
        colorPart = gComparisonDataTemplateP(k)(0)
        If Not aggTemplateP.Exists(styleKey) Then
            aggTemplateP.Add styleKey, colorPart
        Else
            Dim existing As String
            existing = aggTemplateP(styleKey)
            Dim arr() As String, found As Boolean, v As Variant
            arr = Split(existing, ",")
            found = False
            For Each v In arr
                If Trim(v) = colorPart Then
                    found = True
                    Exit For
                End If
            Next v
            If Not found Then
                aggTemplateP(styleKey) = existing & "," & colorPart
            End If
        End If
    Next k
    
    Dim rowCounter As Long
    rowCounter = 2
    Dim templateSizes As String
    templateSizes = Join(gFixedSizes, ",")
    
    Dim style As Variant
    For Each style In gComparisonDataTemplate.Keys
        wsCompare.Cells(rowCounter, "A").Value = style
        wsCompare.Cells(rowCounter, "B").Value = gComparisonDataTemplate(style)(0)
        wsCompare.Cells(rowCounter, "C").Value = gComparisonDataTemplate(style)(1)
        
        If aggTemplateP.Exists(style) Then
            wsCompare.Cells(rowCounter, "D").Value = aggTemplateP(style)
            wsCompare.Cells(rowCounter, "E").Value = templateSizes
            ' Compare aggregated colors (after trimming and sorting alphabetically)
            Dim tmplColors As String, tplpColors As String
            tmplColors = gComparisonDataTemplate(style)(0)
            tplpColors = aggTemplateP(style)
            If StrComp(tmplColors, tplpColors, vbTextCompare) = 0 And _
               gComparisonDataTemplate(style)(1) = templateSizes Then
                wsCompare.Cells(rowCounter, "F").Value = "OK"
            Else
                wsCompare.Cells(rowCounter, "F").Value = "Mismatch"
                LogError "Cross comparison error for SKU " & style & ". Template colors: " & tmplColors & "; Template_P colors: " & tplpColors
            End If
        Else
            wsCompare.Cells(rowCounter, "D").Value = "Not Found"
            wsCompare.Cells(rowCounter, "F").Value = "Missing Template_P Data"
            LogError "Missing Template_P data for SKU " & style & "."
        End If
        rowCounter = rowCounter + 1
    Next style
    wsCompare.Columns.AutoFit
End Sub



'==================================================
' Clean sheet name to remove invalid characters and trim length
'==================================================
Function CleanSheetName(sName As String) As String
    Dim invalidChars As Variant, i As Long
    invalidChars = Array("/", "\", "[", "]", "*", "?", ":")
    For i = LBound(invalidChars) To UBound(invalidChars)
        sName = Replace(sName, invalidChars(i), "_")
    Next i
    If Len(sName) > 31 Then sName = Left(sName, 31)
    CleanSheetName = sName
End Function

'==================================================
' Generate a unique file path based on existing files
'==================================================
Function GenerateUniqueFilePath(folderPath As String, baseName As String, extension As String) As String
    Dim filePath As String, versionNum As Long
    filePath = folderPath & "\" & baseName & extension
    versionNum = 1
    Do While Dir(filePath) <> ""
        filePath = folderPath & "\" & baseName & "_v" & versionNum & extension
        versionNum = versionNum + 1
    Loop
    GenerateUniqueFilePath = filePath
End Function

Sub LogError(errorMsg As String)
    Dim wsError As Worksheet
    On Error Resume Next
    Set wsError = ThisWorkbook.Sheets(ERROR_LOG_SHEET_NAME)
    If wsError Is Nothing Then
        Set wsError = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
        wsError.Name = ERROR_LOG_SHEET_NAME
        With wsError
            .Range("A1").Value = "Timestamp"
            .Range("B1").Value = "Error Message"
        End With
    End If
    On Error GoTo 0
    Dim lastRow As Long
    lastRow = wsError.Cells(wsError.Rows.Count, "A").End(xlUp).Row + 1
    wsError.Cells(lastRow, "A").Value = Now()
    wsError.Cells(lastRow, "B").Value = errorMsg
End Sub

'==============================
' Process a single style (MainSKU) in Template (buy plan)
'==============================
Sub ProcessSKU(ByVal skuKey As String, rowsColl As Collection, wsSource As Worksheet, _
               wsTemplate As Worksheet, colMap As Object, wbOut As Workbook)

    ' (GLOBAL SWITCH) error handling
    If ERROR_HANDLE Then
        On Error GoTo ErrHandler
    Else
        On Error Resume Next
    End If

    If DEBUG_MODE Then Debug.Print "  -> ProcessSKU: " & skuKey & ", #rows= " & rowsColl.Count

    wsTemplate.Copy After:=wbOut.Sheets(wbOut.Sheets.Count)
    Dim wsOut As Worksheet
    Set wsOut = wbOut.Sheets(wbOut.Sheets.Count)
    wsOut.Name = CleanSheetName(CStr(skuKey))

    PopulateFabricDetails wsOut, wsSource, rowsColl(1), colMap
    InsertPhotos wsOut, wsSource, rowsColl(1), colMap
    ProcessDataGrouping wsOut, wsSource, rowsColl, colMap

    wsOut.Range(OUTPUT_DATE_CELL).Value = Format(Date, DATE_FORMAT)

    ' Aggregate color info + total for cross-comparison
    Dim aggColors As String, styleTotal As Double
    aggColors = AggregateColors(rowsColl, wsSource, colMap)
    styleTotal = 0

    Dim r As Variant
    For Each r In rowsColl
        styleTotal = styleTotal + _
                     Val(wsSource.Cells(r, colMap("XS")).Value) + _
                     Val(wsSource.Cells(r, colMap("S")).Value) + _
                     Val(wsSource.Cells(r, colMap("M")).Value) + _
                     Val(wsSource.Cells(r, colMap("L")).Value) + _
                     Val(wsSource.Cells(r, colMap("XL")).Value) + _
                     Val(wsSource.Cells(r, colMap("XXL")).Value)
    Next r

    If Not gComparisonDataTemplate.Exists(skuKey) Then
        gComparisonDataTemplate.Add skuKey, Array(aggColors, Join(gFixedSizes, ","), styleTotal)
    End If

    If DEBUG_MODE Then Debug.Print "    => style=" & skuKey & ", colors=" & aggColors & ", total=" & styleTotal

CleanExit:
    Exit Sub
ErrHandler:
    If ERROR_HANDLE Then
        MsgBox "Error " & Err.Number & ": " & Err.Description, vbExclamation
        If DEBUG_MODE Then Debug.Print "Error in ProcessSKU: " & Err.Number & ": " & Err.Description
        Resume CleanExit
    Else
        Resume Next
    End If
End Sub

'==============================
' Process grouping and summary (buy plan)
'==============================
Sub ProcessDataGrouping(wsOut As Worksheet, wsSource As Worksheet, rowsColl As Collection, colMap As Object)

    If ERROR_HANDLE Then
        On Error GoTo ErrHandler
    Else
        On Error Resume Next
    End If

    If DEBUG_MODE Then Debug.Print "      ProcessDataGrouping (#rows = " & rowsColl.Count & ")"

    Dim fixedSizes As Variant
    fixedSizes = gFixedSizes

    ' Write size headers
    Dim i As Long
    For i = LBound(fixedSizes) To UBound(fixedSizes)
        wsOut.Cells(FIXED_SIZE_HEADER_ROW, FIXED_SIZE_COL_START + i).Value = fixedSizes(i)
    Next i

    ' Group by PONo + ConfigSKU + colorDesc
    Dim dictGroup As Object, rIdx As Variant
    Set dictGroup = CreateObject("Scripting.Dictionary")

    For Each rIdx In rowsColl
        Dim grpKey As String
        grpKey = wsSource.Cells(rIdx, colMap("PONo")).Value & "|" & _
                 wsSource.Cells(rIdx, colMap("ConfigSKU")).Value & "|" & _
                 wsSource.Cells(rIdx, colMap("ColorDesc")).Value
        If Not dictGroup.Exists(grpKey) Then
            Set dictGroup(grpKey) = New Collection
        End If
        dictGroup(grpKey).Add rIdx
    Next rIdx

    If DEBUG_MODE Then Debug.Print "       => Found " & dictGroup.Count & " group(s)."
    Dim groupCount As Long: groupCount = dictGroup.Count

    If groupCount > 1 Then
        Dim newRowsCount As Long
        newRowsCount = groupCount - 1
    
        ' 1. Insert the extra rows (same logic as before)
        wsOut.Rows(GROUP_START_ROW & ":" & GROUP_START_ROW).Resize(newRowsCount).Insert Shift:=xlDown
    
        ' 2. Force correct formatting:
        '    Copy the format of the first data row (GROUP_START_ROW)
        '    onto ALL data rows we now have.
        wsOut.Rows(GROUP_START_ROW + 1).Copy
        wsOut.Rows(GROUP_START_ROW & ":" & (GROUP_START_ROW + newRowsCount)).PasteSpecial xlPasteFormats
        Application.CutCopyMode = False
    End If


    Dim groupIndex As Long, outRow As Long
    groupIndex = 0

    Dim grp As Variant
    For Each grp In dictGroup.Keys
        groupIndex = groupIndex + 1
        outRow = GROUP_START_ROW + (groupIndex - 1)

        If DEBUG_MODE Then Debug.Print "         Group " & groupIndex & " / " & groupCount & ": " & grp
        UpdateControl ThisWorkbook.Sheets(CONTROL_SHEET_NAME), "D7", "Processing group " & groupIndex

        Dim collGroup As Collection
        Set collGroup = dictGroup(grp)

        Dim groupFirstRow As Long
        groupFirstRow = collGroup(1)

        ' Basic info from first row
        With wsOut
            .Cells(outRow, 2).Value = wsSource.Cells(groupFirstRow, colMap("MainSKU")).Value
            .Cells(outRow, 3).Value = wsSource.Cells(groupFirstRow, colMap("Brand")).Value
            .Cells(outRow, 4).Value = wsSource.Cells(groupFirstRow, colMap("ArticleName")).Value
            .Cells(outRow, 5).Value = wsSource.Cells(groupFirstRow, colMap("PONo")).Value
            .Cells(outRow, 6).Value = wsSource.Cells(groupFirstRow, colMap("ConfigSKU")).Value
            .Cells(outRow, 7).Value = wsSource.Cells(groupFirstRow, colMap("ColorDesc")).Value
            .Cells(outRow, 8).Value = wsSource.Cells(groupFirstRow, colMap("ŃŐÉ«")).Value
            .Cells(outRow, 9).Value = wsSource.Cells(groupFirstRow, colMap("Ö÷±ęÉ«")).Value
            .Cells(outRow, 1).Value = wsSource.Cells(groupFirstRow, colMap("şĎÍ¬şĹ")).Value
            .Cells(outRow, 18).Value = wsSource.Cells(groupFirstRow, colMap("Ŕëł§Ę±Ľä")).Value
            .Cells(outRow, 16).Value = wsSource.Cells(groupFirstRow, colMap("´¬ŃůŇŞÇó")).Value
        End With

        ' Re-initialize the sums for each group
        Dim sumXS As Long, sumS As Long, sumM As Long, sumL As Long, sumXL As Long, sumXXL As Long
        sumXS = 0: sumS = 0: sumM = 0: sumL = 0: sumXL = 0: sumXXL = 0

        Dim totalQuantity As Long
        totalQuantity = Val(wsSource.Cells(groupFirstRow, colMap("TOTAL")).Value)
        If DEBUG_MODE Then Debug.Print "            totalQuantity from source: " & totalQuantity

        Dim rowNum As Variant
        For Each rowNum In collGroup
            sumXS = sumXS + Val(wsSource.Cells(rowNum, colMap("XS")).Value)
            sumS = sumS + Val(wsSource.Cells(rowNum, colMap("S")).Value)
            sumM = sumM + Val(wsSource.Cells(rowNum, colMap("M")).Value)
            sumL = sumL + Val(wsSource.Cells(rowNum, colMap("L")).Value)
            sumXL = sumXL + Val(wsSource.Cells(rowNum, colMap("XL")).Value)
            sumXXL = sumXXL + Val(wsSource.Cells(rowNum, colMap("XXL")).Value)
        Next rowNum

        Dim summed As Long
        summed = sumXS + sumS + sumM + sumL + sumXL + sumXXL

        ' Check mismatch
        If summed <> totalQuantity Then
            LogError "Mismatch for PONo: " & wsSource.Cells(groupFirstRow, colMap("PONo")).Value & _
                     " expected " & totalQuantity & ", got " & summed
            If DEBUG_MODE Then Debug.Print "            => MISMATCH: expected " & totalQuantity & ", got " & summed
        Else
            If DEBUG_MODE Then Debug.Print "            => Sums match: " & summed
        End If

        ' Put sums
        wsOut.Cells(outRow, FIXED_SIZE_COL_START).Value = sumXS
        wsOut.Cells(outRow, FIXED_SIZE_COL_START + 1).Value = sumS
        wsOut.Cells(outRow, FIXED_SIZE_COL_START + 2).Value = sumM
        wsOut.Cells(outRow, FIXED_SIZE_COL_START + 3).Value = sumL
        wsOut.Cells(outRow, FIXED_SIZE_COL_START + 4).Value = sumXL
        wsOut.Cells(outRow, FIXED_SIZE_COL_START + 5).Value = sumXXL

        wsOut.Cells(outRow, TOTAL_FORMULA_COL).Formula = "=SUM(" & _
            wsOut.Cells(outRow, FIXED_SIZE_COL_START).Address(False, False) & ":" & _
            wsOut.Cells(outRow, FIXED_SIZE_COL_START + 5).Address(False, False) & ")"
    Next grp

    Dim lastGroupRow As Long, overallRow As Long
    lastGroupRow = GROUP_START_ROW + groupCount - 1
    overallRow = lastGroupRow + 1

    wsOut.Cells(overallRow, TOTAL_LABEL_COL).Value = "Total:"
    wsOut.Cells(overallRow, TOTAL_FORMULA_COL).Formula = "=SUM(" & _
        wsOut.Cells(GROUP_START_ROW, TOTAL_FORMULA_COL).Address(False, False) & ":" & _
        wsOut.Cells(lastGroupRow, TOTAL_FORMULA_COL).Address(False, False) & ")"

    wsOut.Range("Q5").Formula = "=SUM(" & _
        wsOut.Cells(GROUP_START_ROW, TOTAL_FORMULA_COL).Address(False, False) & ":" & _
        wsOut.Cells(lastGroupRow, TOTAL_FORMULA_COL).Address(False, False) & ")"

    UpdateControl ThisWorkbook.Sheets(CONTROL_SHEET_NAME), "D7", "Completed processing groups."
    If DEBUG_MODE Then Debug.Print "      Done with ProcessDataGrouping."

CleanExit:
    Exit Sub
ErrHandler:
    If ERROR_HANDLE Then
        MsgBox "Error " & Err.Number & ": " & Err.Description, vbExclamation
        If DEBUG_MODE Then Debug.Print "Error in ProcessDataGrouping: " & Err.Number & ": " & Err.Description
        Resume CleanExit
    Else
        Resume Next
    End If
End Sub

'==============================
' Aggregate distinct colors ...
'==============================
Function AggregateColors(rowsColl As Collection, ws As Worksheet, colMap As Object) As String

    If ERROR_HANDLE Then
        On Error GoTo ErrHandler
    Else
        On Error Resume Next
    End If

    Dim dictColors As Object
    Set dictColors = CreateObject("Scripting.Dictionary")

    Dim r As Variant, colorVal As String
    For Each r In rowsColl
        colorVal = Trim(ws.Cells(r, colMap("ŃŐÉ«")).Value)
        If colorVal <> "" Then
            If Not dictColors.Exists(colorVal) Then dictColors.Add colorVal, True
        End If
    Next r

    AggregateColors = Join(dictColors.Keys, ",")
    If DEBUG_MODE Then Debug.Print "      AggregateColors => " & AggregateColors

CleanExit:
    Exit Function
ErrHandler:
    If ERROR_HANDLE Then
        If DEBUG_MODE Then Debug.Print "Error in AggregateColors: " & Err.Number & ": " & Err.Description
        Resume CleanExit
    Else
        Resume Next
    End If
End Function

'==============================
' NormalizePath
' - Removes any trailing "\" or "/" (but keeps "C:\" intact)
'==============================
Private Function NormalizePath(ByVal p As String) As String
    Dim orig As String
    orig = p

    ' replace "/" with "\" first
    p = Replace(p, "/", "\")

    ' special case: if it's just like "C:\" then leave it
    If Len(p) = 3 And Mid$(p, 2, 2) = ":\\" Then
        NormalizePath = p
        Exit Function
    End If

    ' otherwise trim any trailing "\" characters
    Do While Len(p) > 0 And Right$(p, 1) = "\"
        p = Left$(p, Len(p) - 1)
    Loop

    NormalizePath = p
End Function

'==============================
' EnsureFolderExistsRecursive
' - Creates the entire folder chain, like mkdir -p
'==============================
Private Sub EnsureFolderExistsRecursive(ByVal fullPath As String)
    Dim parts() As String
    Dim buildPath As String
    Dim i As Long

    ' normalise slashes
    fullPath = Replace(fullPath, "/", "\")

    ' trim trailing "\" unless it's just "C:\"
    If Not (Len(fullPath) = 3 And Mid$(fullPath, 2, 2) = ":\") Then
        Do While Len(fullPath) > 0 And Right$(fullPath, 1) = "\"
            fullPath = Left$(fullPath, Len(fullPath) - 1)
        Loop
    End If

    parts = Split(fullPath, "\")

    If UBound(parts) < 0 Then Exit Sub

    ' handle drive root
    buildPath = parts(0)
    If InStr(buildPath, ":") > 0 And Right$(buildPath, 1) <> "\" Then
        buildPath = buildPath & "\"
    End If

    ' walk down each segment and create if missing
    For i = 1 To UBound(parts)
        If Right$(buildPath, 1) = "\" Then
            buildPath = buildPath & parts(i)
        Else
            buildPath = buildPath & "\" & parts(i)
        End If

        If Dir(buildPath, vbDirectory) = "" Then
            MkDir buildPath
        End If
    Next i
End Sub

'==============================
' CompareColorSets
'==============================
Function CompareColorSets(str1 As String, str2 As String) As Boolean

    If ERROR_HANDLE Then
        On Error GoTo ErrHandler
    Else
        On Error Resume Next
    End If

    Dim arr1() As String, arr2() As String
    arr1 = Split(Replace(Replace(str1, " ", ""), ",,", ","), ",")
    arr2 = Split(Replace(Replace(str2, " ", ""), ",,", ","), ",")

    QuickSort arr1, LBound(arr1), UBound(arr1)
    QuickSort arr2, LBound(arr2), UBound(arr2)

    Dim joined1 As String, joined2 As String
    joined1 = Join(arr1, ",")
    joined2 = Join(arr2, ",")

    CompareColorSets = (StrComp(joined1, joined2, vbTextCompare) = 0)

CleanExit:
    Exit Function
ErrHandler:
    If ERROR_HANDLE Then
        If DEBUG_MODE Then Debug.Print "Error in CompareColorSets: " & Err.Number & ": " & Err.Description
        Resume CleanExit
    Else
        Resume Next
    End If
End Function

'==============================
' QuickSort for color sets
'==============================
Sub QuickSort(arr() As String, ByVal first As Long, ByVal last As Long)

    If ERROR_HANDLE Then
        On Error GoTo ErrHandler
    Else
        On Error Resume Next
    End If

    Dim pivot As String, tmp As String
    Dim i As Long, j As Long

    If first >= last Then GoTo CleanExit
    pivot = arr((first + last) \ 2)
    i = first
    j = last
    Do While i <= j
        Do While StrComp(arr(i), pivot, vbTextCompare) < 0
            i = i + 1
        Loop
        Do While StrComp(arr(j), pivot, vbTextCompare) > 0
            j = j - 1
        Loop
        If i <= j Then
            tmp = arr(i)
            arr(i) = arr(j)
            arr(j) = tmp
            i = i + 1
            j = j - 1
        End If
    Loop
    If first < j Then QuickSort arr, first, j
    If i < last Then QuickSort arr, i, last

CleanExit:
    Exit Sub
ErrHandler:
    If ERROR_HANDLE Then
        If DEBUG_MODE Then Debug.Print "Error in QuickSort: " & Err.Number & ": " & Err.Description
        Resume CleanExit
    Else
        Resume Next
    End If
End Sub

'==============================
' Generate Unique File Path
'==============================
Function GenerateUniqueFilePath1(folderPath As String, baseName As String, extension As String) As String

    If ERROR_HANDLE Then
        On Error GoTo ErrHandler
    Else
        On Error Resume Next
    End If

    Dim filePath As String, versionNum As Long
    filePath = folderPath & "\" & baseName & extension
    versionNum = 1
    Do While Dir(filePath) <> ""
        filePath = folderPath & "\" & baseName & "_v" & versionNum & extension
        versionNum = versionNum + 1
    Loop
    GenerateUniqueFilePath = filePath

CleanExit:
    Exit Function
ErrHandler:
    If ERROR_HANDLE Then
        If DEBUG_MODE Then Debug.Print "Error in GenerateUniqueFilePath: " & Err.Number & ": " & Err.Description
        Resume CleanExit
    Else
        Resume Next
    End If
End Function
Function GenerateUniqueFilePath(ByVal folderPath As String, _
                                ByVal baseName As String, _
                                ByVal extension As String) As String
    Dim filePath As String
    Dim versionNum As Long
    Dim test As String

    folderPath = Trim$(folderPath)
    baseName = Trim$(baseName)
    extension = Trim$(extension)

    ' Basic guardrails
    If folderPath = "" Then Err.Raise vbObjectError + 200, , "folderPath is empty."
    If baseName = "" Then Err.Raise vbObjectError + 201, , "baseName is empty."
    If Left$(extension, 1) <> "." Then extension = "." & extension

    ' Ensure folderPath has no trailing "\" (avoid double slashes)
    Do While Right$(folderPath, 1) = "\" Or Right$(folderPath, 1) = "/"
        folderPath = Left$(folderPath, Len(folderPath) - 1)
    Loop

    filePath = folderPath & "\" & baseName & extension
    versionNum = 1

    Do
        On Error GoTo DirErr
        test = Dir$(filePath)
        On Error GoTo 0

        If test = "" Then Exit Do  ' file does not exist -> OK

        versionNum = versionNum + 1
        If versionNum > 9999 Then Err.Raise vbObjectError + 202, , "Too many versions. Check folder/path."

        filePath = folderPath & "\" & baseName & "_v" & (versionNum - 1) & extension
    Loop

    GenerateUniqueFilePath = filePath
    Exit Function

DirErr:
    ' If Dir fails, fail fast with a useful message instead of looping forever
    Err.Raise vbObjectError + 203, , "Dir() failed for path: " & filePath & " | " & Err.Description
End Function

Sub LogError(errorMsg As String)

    If ERROR_HANDLE Then
        On Error GoTo ErrHandler
    Else
        On Error Resume Next
    End If

    Dim wsError As Worksheet
    Set wsError = ThisWorkbook.Sheets(ERROR_LOG_SHEET_NAME)
    If wsError Is Nothing Then
        Set wsError = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
        wsError.Name = ERROR_LOG_SHEET_NAME
        With wsError
            .Range("A1").Value = "Timestamp"
            .Range("B1").Value = "Error Message"
        End With
    End If

    Dim lastRow As Long
    lastRow = wsError.Cells(wsError.Rows.Count, "A").End(xlUp).Row + 1
    wsError.Cells(lastRow, "A").Value = Now()
    wsError.Cells(lastRow, "B").Value = errorMsg

    If DEBUG_MODE Then Debug.Print "[ERROR LOG] " & errorMsg

CleanExit:
    Exit Sub
ErrHandler:
    If ERROR_HANDLE Then
        If DEBUG_MODE Then Debug.Print "Error in LogError: " & Err.Number & ": " & Err.Description
        Resume CleanExit
    Else
        Resume Next
    End If
End Sub

'==============================
' CleanSheetName
'==============================
Function CleanSheetName(sName As String) As String

    If ERROR_HANDLE Then
        On Error GoTo ErrHandler
    Else
        On Error Resume Next
    End If

    Dim invalidChars As Variant, i As Long
    invalidChars = Array("/", "\", "[", "]", "*", "?", ":")
    For i = LBound(invalidChars) To UBound(invalidChars)
        sName = Replace(sName, invalidChars(i), "_")
    Next i
    If Len(sName) > 31 Then sName = Left(sName, 31)
    CleanSheetName = sName

CleanExit:
    Exit Function
ErrHandler:
    If ERROR_HANDLE Then
        If DEBUG_MODE Then Debug.Print "Error in CleanSheetName: " & Err.Number & ": " & Err.Description
        Resume CleanExit
    Else
        Resume Next
    End If
End Function
